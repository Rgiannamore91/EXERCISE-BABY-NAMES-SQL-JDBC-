SELECT *
FROM babynames
WHERE US_STATE = 'MD'
  AND DATE_YEAR = 1991
  AND GENDER ='M'
ORDER BY NUM_BABIES DESC
    FETCH FIRST 1 ROWS ONLY;

SELECT NAME
FROM babynames
WHERE US_STATE = 'MD'
  AND DATE_YEAR = 1991
  AND GENDER ='M'
ORDER BY NUM_BABIES DESC
FETCH FIRST 1 ROWS ONLY;

SELECT DATE_YEAR
FROM babynames
WHERE NAME = 'Christopher' AND GENDER = 'M'
GROUP BY DATE_YEAR
ORDER BY COUNT(*) DESC
FETCH FIRST 1 ROWS ONLY;

SELECT DATE_YEAR
FROM babynames

GROUP BY DATE_YEAR
